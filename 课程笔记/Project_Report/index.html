    DataPath设计¶

    根据数据通路图：

    分别写出对应的模块，然后在Core.sv中连接起来

    • PC

    简单的时序电路，使pc更新为next_pc

    module PC (
    input clk,
    input rst,
    input CorePack::data_t next_pc,
    output CorePack::data_t pc
);
    import CorePack::*;
    always @(posedge clk or posedge rst) begin
        if(rst) pc <= 0;
        else pc <= next_pc;
    end
endmodule
    • RegFile

    寄存器组，write使用时序逻辑，当使能时在对应位置写入，read直接用组合电路连出即可

      import CorePack::*;
  integer i;
  data_t register [1:31]; // x1 - x31, x0 keeps zero
  //取出对应寄存器的值
  assign read_data_1 = (read_addr_1 == 0) ? 0 : register[read_addr_1];
  assign read_data_2 = (read_addr_2 == 0) ? 0 : register[read_addr_2];

  always @(posedge clk or posedge rst)begin
    if(rst) 
      for(i=1;i<=31;i=i+1) register[i] <= 0;
    else if(we && write_addr != 0) register[write_addr] <= write_data; //当we为1且寄存器不为x0时，写入

  end
endmodule
    • ALU

    根据alu_op的值选择对应的运算即可

      always_comb begin
    tmp = 0; //添加默认值防止latch
    case(alu_op)
      ALU_ADD:  res = a + b;  //加法
      ALU_SUB:  res = a - b;  //减法
      ALU_AND:  res = a & b;  //与
      ALU_OR:   res = a | b;  //或
      ALU_XOR:  res = a ^ b;  //异或
      ALU_SLT:  res = ($signed(a) < $signed(b)) ? 1 : 0;  //有符号比较 是否小于
      ALU_SLTU: res = (a < b) ? 1 : 0;  //无符号比较 是否小于
      ALU_SLL:  res = a << b[5:0];  //a左移b位
      ALU_SRL:  res = a >> b[5:0];  //a逻辑右移b位
      ALU_SRA:  res = $signed(a) >>> b[5:0];  //a算术右移b位
      ALU_ADDW: begin
        tmp = a[31:0] + b[31:0];
        res = {{32{tmp[31]}},tmp};  // 32位加法结果符号扩展到64位
      end  
      ALU_SUBW: begin
        tmp = a[31:0] - b[31:0];
        res = {{32{tmp[31]}},tmp}; // 32位减法结果符号扩展到64位
      end  
      ALU_SLLW: begin
        tmp = a[31:0] << b[4:0];
        res = {{32{tmp[31]}},tmp}; // 32位左移结果符号扩展到64位
      end 
      ALU_SRLW: begin
        tmp = a[31:0] >> b[4:0];
        res = {{32{tmp[31]}},tmp}; // 32位逻辑右移结果符号扩展到64位
      end
      ALU_SRAW: res = {{32{a[31]}}, $signed(a[31:0]) >>> b[4:0]}; // 32位算术右移结果符号扩展到64位
      ALU_DEFAULT: res = 0;
      default:  res = 0;
    endcase
  end

    • Data Truncation

    把read出来的data根据类型和偏移位进行调整，主要分为截取和扩展两个部分

    always_comb begin
    case (mem_op)
         MEM_D:  mem_tmp = rdata;
         MEM_W:  mem_tmp =(rdata >> offset)&64'hFFFFFFFF;
         MEM_H:  mem_tmp =(rdata >> offset) & 64'hFFFF;
         MEM_B:  mem_tmp =(rdata >> offset) & 64'hFF;
         MEM_UB: mem_tmp =(rdata >> offset) & 64'hFF;
         MEM_UH: mem_tmp =(rdata >> offset) & 64'hFFFF;
         MEM_UW: mem_tmp =(rdata >> offset)&64'hFFFFFFFF;
            default: mem_tmp = 64'h0;
        endcase
    case (mem_op)
         MEM_D:  extended_mem = mem_tmp; 
         MEM_W:  extended_mem = {{32{mem_tmp[31]}}, mem_tmp[31:0]};  
         MEM_H:  extended_mem = {{48{mem_tmp[15]}}, mem_tmp[15:0]};  
         MEM_B:  extended_mem = {{56{mem_tmp[7]}}, mem_tmp[7:0]};    
         MEM_UB: extended_mem = {56'b0, mem_tmp[7:0]};   
         MEM_UH: extended_mem = {48'b0, mem_tmp[15:0]};   
         MEM_UW: extended_mem = {32'b0, mem_tmp[31:0]};   
         default: extended_mem = 64'h0;
        endcase
    end

    assign mem = extended_mem;

    Controller设计¶

    我使用的控制信号如下：

    module controller (
    input CorePack::inst_t inst,
    output we_reg,
    output we_mem,
    output re_mem,
    output is_b,
    output is_j,
    output CorePack::imm_op_enum immgen_op,
    output CorePack::alu_op_enum alu_op,
    output CorePack::cmp_op_enum cmp_op,
    output alu_asel,
    output alu_bsel,
    output alu_lui,
    output CorePack::wb_sel_op_enum wb_sel,
    output CorePack::mem_op_enum mem_op
);

    相比实验指导，我把alu_asel和alu_bsel改为一位，便于二段译码，同时新增alu_lui信号，用来判断lui指令，给alu的a口输入0值。

    • 对所有一位宽信号进行二段译码
        logic inst_load  =   opcode == LOAD_OPCODE;
    logic inst_reg   =   opcode == REG_OPCODE;
    logic inst_imm   =   opcode == IMM_OPCODE;
    logic inst_store =   opcode == STORE_OPCODE;
    logic inst_auipc =   opcode == AUIPC_OPCODE;
    logic inst_immw  =   opcode == IMMW_OPCODE;
    logic inst_lui   =   opcode == LUI_OPCODE;
    logic inst_regw  =   opcode == REGW_OPCODE;
    logic inst_branch=   opcode == BRANCH_OPCODE;
    logic inst_jalr  =   opcode == JALR_OPCODE;
    logic inst_jal   =   opcode == JAL_OPCODE;

    assign is_b   = inst_branch; 
    assign is_j   = inst_jal | inst_jalr;
    assign we_reg = inst_reg | inst_regw | inst_load | inst_imm | inst_immw | inst_auipc | inst_lui | inst_jal | inst_jalr;
    assign we_mem = inst_store;
    assign re_mem = inst_load;
    assign alu_asel = inst_auipc | inst_branch | inst_jal ; //asel = pc
    assign alu_bsel = inst_auipc | inst_branch | inst_imm | inst_immw |inst_lui | inst_jal | inst_jalr | inst_store | inst_load; //bsel = imm
    assign alu_lui  = inst_lui;
